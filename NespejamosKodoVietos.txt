const searchInput = document.getElementById("searchInput");
const resultsDiv = document.getElementById("results");
const statusText = document.getElementById("status");

// API - paprasta nuoroda i serverius kurie grazina informacija
const allShowsApi = "https://api.tvmaze.com/shows";
const searchShows = "https://api.tvmaze.com/search/shows?q=";

function getAllShows() {
  fetch(allShowsApi)
    // resolve
    .then((response) => {
      // .json() irgi yra Promise
      return response.json();
    })
    .then((data) => {
      // data - jau is response paverstas i normalu json pavidala informacija
      console.log(data);
    })
    // resolve
    // reject
    .catch((error) => {
      console.log(error);
    });
  // reject
}

// ant puslapio uzkrovimo paleis sita funkcija
getAllShows();



searchInput.addEventListener("input", () => {
  const value = searchInput.value;

  if (value.length >= 2) {
    searchShowsByName(value);
  }
});

function searchShowsByName(value) {
  statusText.textContent = "ieskoma...";
  resultsDiv.innerHTML = "";

  fetch(`${searchShows}${value}`)
    .then((response) => {
      return response.json();
    })
    .then((data) => {
      const movies = data.map((item) => item.show);

      console.log(movies);
    });
}

function renderMovies(movies) {
  movies.forEach((movie) => {
    const div = document.createElement("div");
    div.className = "card";
    const rating = movie.rating.average ?? "Vertinimo nera";
    const image = movie.image?.medium ?? "";

    div.innerHTML = `
        ${image ? `<img src="${image}" alt="nuotrauka">` : ""}
        <h3>${movie.name}</h3>
        <h3>${rating}</h3>
        `;

    resultsDiv.appendChild(div);
  });
}
